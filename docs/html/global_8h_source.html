72 static inline void sleep_for_microseconds(unsigned int us) {
73  T5CON = 0x8000; // enable Timer5, source PBCLK, 1:1 prescaler
74  // delay 100us per loop until less than 100us remain
75  while( us >= 100) {
76  us-=100;
77  TMR5 = 0;
78  while( TMR5 < PERIPH_CLK/10000);
79  }
80  // delay 10us per loop until less than 10us remain
81  while( us >= 10) {
82  us-=10;
83  TMR5 = 0;
84  while( TMR5 < PERIPH_CLK/100000);
85  }
86  // delay 1us per loop until finished
87  while( us > 0) {
88  us--;
89  TMR5 = 0;
90  while( TMR5 < PERIPH_CLK/1000000);
91  }
92  // turn off Timer5 so function is self-contained
93  T5CONCLR = 0x8000;
94 }

179 static inline void SSN_LED_INDICATE(uint8_t this_state) {
180  /* This one function will determine the state of SSN and give the proper LED heartbeat indication */
181 
182  switch(this_state) {
183 
184  /* No mac address? */
185  case NO_MAC_STATE:
186  Node_Up_Not_Configured_LED_INDICATE();
187  break;
188 
189  /* No configuration? */
190  case NO_CONFIG_STATE:
191  Node_Up_Not_Configured_LED_INDICATE();
192  break;
193 
194  /* No time of day? */
195  case NO_TIMEOFDAY_STATE:
196  Node_Up_Not_Configured_LED_INDICATE();
197  break;
198 
199  /* CONFIG received so ack config? */
200  case ACK_CONFIG_STATE:
201  Node_Up_Not_Configured_LED_INDICATE();
202  break;
203 
204  /* there is no ethernet ? */
205  case NO_ETHERNET_STATE:
206  No_Ethernet_LED_INDICATE();
207  break;
208 
209  /* self tests have failed, so... */
210  case SELF_TEST_FAILED_STATE:
211  Self_Test_Failed_LED_INDICATE();
212  break;
213 
214  /* there is no ethernet ? */
215  case NO_CURRENT_SENSOR_STATE:
216  Current_Sensors_Disconnected_LED_INDICATE();
217  break;
218 
219  /* self tests have failed, so... */
220  case ABNORMAL_ACTIVITY_STATE:
221  Abnormal_Activity_LED_INDICATE();
222  break;
223 
224  /* status update message? */
225  case NORMAL_ACTIVITY_STATE:
226  Normal_Operation_LED_INDICATE();
227  break;
228 
229  default:
230  break;
231  }
232 }
